#### 3.2.1 截图自动检测与导入

```swift
import Photos
import SwiftUI

class ScreenshotMonitor: ObservableObject {
    private var fetchResult: PHFetchResult<PHAsset>?
    private var lastScreenshotCount: Int = 0
    @Published var newScreenshots: [ScreenshotItem] = []
    
    func startMonitoring() {
        let options = PHFetchOptions()
        options.predicate = NSPredicate(
            format: "mediaType == %d AND isScreenshot == YES",
            PHAssetMediaType.image.rawValue
        )
        options.sortDescriptors = [NSSortDescriptor(key: "creationDate", ascending: false)]
        
        fetchResult = PHAsset.fetchAssets(with: options)
        lastScreenshotCount = fetchResult?.count ?? 0
        
        PHPhotoLibrary.shared().register(self)
    }
    
    func fetchNewScreenshots() -> [PHAsset] {
        let options = PHFetchOptions()
        options.predicate = NSPredicate(
            format: "mediaType == %d AND isScreenshot == YES",
            PHAssetMediaType.image.rawValue
        )
        options.sortDescriptors = [NSSortDescriptor(key: "creationDate", ascending: false)]
        
        let result = PHAsset.fetchAssets(with: options)
        var assets: [PHAsset] = []
        result.enumerateObjects { asset, _, _ in
            assets.append(asset)
        }
        return assets
    }
}

extension ScreenshotMonitor: PHPhotoLibraryChangeObserver {
    func photoLibraryDidChange(_ changeInstance: PHChange) {
        guard let fetchResult = fetchResult,
              let details = changeInstance.changeDetails(for: fetchResult) else { return }
        
        let newCount = details.fetchResultAfterChanges.count
        if newCount > lastScreenshotCount {
            let inserted = details.insertedObjects
            DispatchQueue.main.async {
                for asset in inserted {
                    self.processNewScreenshot(asset)
                }
                self.lastScreenshotCount = newCount
            }
        }
    }
    
    private func processNewScreenshot(_ asset: PHAsset) {
        let processor = ScreenshotProcessor()
        Task {
            let item = await processor.analyze(asset: asset)
            await MainActor.run {
                self.newScreenshots.append(item)
            }
        }
    }
}
```

#### 3.2.2 AI自动分类引擎

```swift
import Vision
import CoreML
import UIKit

enum ScreenshotCategory: String, CaseIterable, Codable {
    case otp = "OTP & Verification"
    case qrCode = "QR Codes"
    case delivery = "Delivery & Tracking"
    case receipt = "Receipts & Payments"
    case social = "Social Media"
    case notes = "Notes & Text"
    case shopping = "Shopping & Products"
    case travel = "Travel & Maps"
    case food = "Recipes & Food"
    case work = "Work & Documents"
    case meme = "Memes & Fun"
    case health = "Health & Fitness"
    case other = "Other"
    
    var icon: String {
        switch self {
        case .otp: return "lock.shield"
        case .qrCode: return "qrcode"
        case .delivery: return "shippingbox"
        case .receipt: return "receipt"
        case .social: return "message"
        case .notes: return "note.text"
        case .shopping: return "bag"
        case .travel: return "map"
        case .food: return "fork.knife"
        case .work: return "briefcase"
        case .meme: return "face.smiling"
        case .health: return "heart.text.square"
        case .other: return "square.grid.2x2"
        }
    }
    
    var isTemporary: Bool {
        [.otp, .qrCode, .delivery].contains(self)
    }
}

class ScreenshotClassifier {
    private let textClassifier: TextBasedClassifier
    private let imageClassifier: ImageBasedClassifier
    
    init() {
        self.textClassifier = TextBasedClassifier()
        self.imageClassifier = ImageBasedClassifier()
    }
    
    func classify(image: UIImage, extractedText: String) async -> ScreenshotCategory {
        if isOTP(text: extractedText) { return .otp }
        if isQRCode(image: image) { return .qrCode }
        if isDeliveryTracking(text: extractedText) { return .delivery }
        if isReceipt(text: extractedText) { return .receipt }
        
        let textCategory = textClassifier.classify(text: extractedText)
        let imageCategory = await imageClassifier.classify(image: image)
        
        return mergeCategories(text: textCategory, image: imageCategory)
    }
    
    private func isOTP(text: String) -> Bool {
        let otpPatterns = [
            "\\b\\d{4,8}\\b",
            "(?i)verification\\s*code",
            "(?i)OTP",
            "(?i)confirm(?:ation)?\\s*code",
            "(?i)security\\s*code",
            "(?i)authentication\\s*code"
        ]
        return otpPatterns.contains { text.range(of: $0, options: .regularExpression) != nil }
    }
    
    private func isQRCode(image: UIImage) -> Bool {
        let request = VNDetectBarcodesRequest()
        request.symbologies = [.qr, .aztec, .pdf417]
        let handler = VNImageRequestHandler(image: image, options: [:])
        try? handler.perform([request])
        return !(request.results?.isEmpty ?? true)
    }
    
    private func isDeliveryTracking(text: String) -> Bool {
        let patterns = [
            "(?i)tracking\\s*(?:number|code|id)",
            "(?i)order\\s*(?:status|confirmed|shipped|delivered)",
            "(?i)estimated\\s*delivery",
            "(?i)1Z[A-Z0-9]{16}",
            "(?i)FedEx|UPS|USPS|DHL"
        ]
        return patterns.contains { text.range(of: $0, options: .regularExpression) != nil }
    }
    
    private func isReceipt(text: String) -> Bool {
        let patterns = [
            "(?i)total\\s*:\\s*\\$",
            "(?i)subtotal|sub\\s*total",
            "(?i)payment\\s*(?:method|received)",
            "(?i)receipt|invoice",
            "\\$\\d+\\.\\d{2}"
        ]
        return patterns.filter { text.range(of: $0, options: .regularExpression) != nil }.count >= 2
    }
    
    private func mergeCategories(text: ScreenshotCategory?, image: ScreenshotCategory?) -> ScreenshotCategory {
        if let text = text { return text }
        if let image = image { return image }
        return .other
    }
}

class TextBasedClassifier {
    private let categoryKeywords: [ScreenshotCategory: [String]] = [
        .social: ["instagram", "twitter", "tiktok", "facebook", "snapchat", "whatsapp", "message from", "dm me"],
        .shopping: ["add to cart", "checkout", "price", "discount", "sale", "buy now", "shop", "order", "$"],
        .travel: ["flight", "hotel", "booking", "reservation", "boarding pass", "gate", "terminal", "airbnb"],
        .food: ["recipe", "ingredients", "cook", "bake", "minutes", "oven", "tablespoon", "cup of"],
        .work: ["meeting", "deadline", "project", "sprint", "jira", "slack", "email", "report"],
        .health: ["doctor", "appointment", "medication", "blood pressure", "calories", "workout"],
        .meme: ["lol", "lmao", "bruh", "when you", "nobody:", "me:"]
    ]
    
    func classify(text: String) -> ScreenshotCategory? {
        let lowercased = text.lowercased()
        var bestMatch: ScreenshotCategory?
        var bestScore = 0
        
        for (category, keywords) in categoryKeywords {
            let score = keywords.filter { lowercased.contains($0) }.count
            if score > bestScore {
                bestScore = score
                bestMatch = category
            }
        }
        return bestMatch
    }
}

class ImageBasedClassifier {
    func classify(image: UIImage) async -> ScreenshotCategory? {
        guard let cgImage = image.cgImage else { return nil }
        
        return await withCheckedContinuation { continuation in
            let request = VNClassifyImageRequest { request, error in
                guard let observations = request.results as? [VNClassificationObservation] else {
                    continuation.resume(returning: nil)
                    return
                }
                
                let topObservations = observations.prefix(5)
                let categories = topObservations.compactMap { obs -> ScreenshotCategory? in
                    self.mapVisionCategory(obs.identifier, confidence: obs.confidence)
                }
                continuation.resume(returning: categories.first)
            }
            
            let handler = VNImageRequestHandler(cgImage: cgImage, options: [:])
            try? handler.perform([request])
        }
    }
    
    private func mapVisionCategory(_ identifier: String, confidence: Float) -> ScreenshotCategory? {
        guard confidence > 0.3 else { return nil }
        if identifier.contains("food") || identifier.contains("recipe") { return .food }
        if identifier.contains("map") || identifier.contains("aerial") { return .travel }
        if identifier.contains("receipt") || identifier.contains("document") { return .receipt }
        return nil
    }
}
```

```swift
import Vision
import UIKit

class OCREngine {
    func extractText(from image: UIImage) async -> String {
        guard let cgImage = image.cgImage else { return "" }
        
        return await withCheckedContinuation { continuation in
            let request = VNRecognizeTextRequest { request, error in
                guard let observations = request.results as? [VNRecognizedTextObservation] else {
                    continuation.resume(returning: "")
                    return
                }
                
                let text = observations.compactMap { observation in
                    observation.topCandidates(1).first?.string
                }.joined(separator: "\n")
                
                continuation.resume(returning: text)
            }
            
            request.recognitionLevel = .accurate
            request.recognitionLanguages = ["en-US", "en-GB", "zh-Hans", "zh-Hant", "ja", "ko", "es", "de", "fr"]
            request.usesLanguageCorrection = true
            
            let handler = VNImageRequestHandler(cgImage: cgImage, options: [:])
            try? handler.perform([request])
        }
    }
    
    func extractText(from asset: PHAsset) async -> String {
        guard let image = await loadImage(from: asset) else { return "" }
        return await extractText(from: image)
    }
    
    private func loadImage(from asset: PHAsset) async -> UIImage? {
        let options = PHImageRequestOptions()
        options.deliveryMode = .highQualityFormat
        options.isSynchronous = false
        
        return await withCheckedContinuation { continuation in
            PHImageManager.default().requestImage(
                for: asset,
                targetSize: CGSize(width: 2048, height: 2048),
                contentMode: .aspectFit,
                options: options
            ) { image, _ in
                continuation.resume(returning: image)
            }
        }
    }
}
```

```swift
import Foundation
import UserNotifications

class AutoCleanEngine {
    private let cleanupRules: [ScreenshotCategory: CleanupRule] = [
        .otp: CleanupRule(hours: 1, name: "OTP codes expire quickly"),
        .qrCode: CleanupRule(hours: 24, name: "QR codes are one-time use"),
        .delivery: CleanupRule(hours: 168, name: "Delivery tracking expires after delivery")
    ]
    
    struct CleanupRule {
        let hours: Int
        let name: String
        var deadline: Date {
            Calendar.current.date(byAdding: .hour, value: hours, to: Date())!
        }
    }
    
    func scheduleCleanupReminder(for item: ScreenshotItem) {
        guard let rule = cleanupRules[item.category] else { return }
        
        let content = UNMutableNotificationContent()
        content.title = "Time to Clean Up 🧹"
        content.body = "Your \(item.category.rawValue) screenshot is \(rule.hours) hours old. Delete it?"
        content.sound = .default
        content.categoryIdentifier = "CLEANUP_ACTION"
        content.userInfo = ["screenshotId": item.id.uuidString]
        
        let trigger = UNTimeIntervalNotificationTrigger(
            timeInterval: TimeInterval(rule.hours * 3600),
            repeats: false
        )
        
        let request = UNNotificationRequest(
            identifier: "cleanup-\(item.id.uuidString)",
            content: content,
            trigger: trigger
        )
        
        UNUserNotificationCenter.current().add(request)
    }
    
    func performBulkCleanup(modelContext: ModelContext) {
        let now = Date()
        let descriptor = FetchDescriptor<ScreenshotItem>(
            predicate: #Predicate { $0.isTemporary && $0.autoDeleteDate != nil && $0.autoDeleteDate! < now && !$0.isDeleted }
        )
        
        do {
            let expired = try modelContext.fetch(descriptor)
            for item in expired {
                deleteScreenshotFromPhotos(item)
                item.isDeleted = true
            }
            try modelContext.save()
        } catch {
            print("Cleanup error: \(error)")
        }
    }
    
    private func deleteScreenshotFromPhotos(_ item: ScreenshotItem) {
        let fetchResult = PHAsset.fetchAssets(
            withLocalIdentifiers: [item.assetLocalIdentifier],
            options: nil
        )
        if let asset = fetchResult.firstObject {
            PHPhotoLibrary.shared().performChanges {
                PHAssetChangeRequest.deleteAssets([asset] as NSFastEnumeration)
            }
        }
    }
}
```

```swift
import SwiftUI

class ScreenshotSearchEngine {
    private var modelContext: ModelContext
    
    init(modelContext: ModelContext) {
        self.modelContext = modelContext
    }
    
    func search(query: String) -> [ScreenshotItem] {
        let descriptor = FetchDescriptor<ScreenshotItem>(
            predicate: #Predicate { !$0.isDeleted }
        )
        
        guard let allItems = try? modelContext.fetch(descriptor) else { return [] }
        
        let lowerQuery = query.lowercased()
        
        return allItems.filter { item in
            item.extractedText.lowercased().contains(lowerQuery) ||
            item.summary.lowercased().contains(lowerQuery) ||
            item.tags.contains(where: { $0.lowercased().contains(lowerQuery) }) ||
            item.category.rawValue.lowercased().contains(lowerQuery)
        }.sorted { $0.createdAt > $1.createdAt }
    }
    
    func searchByCategory(_ category: ScreenshotCategory) -> [ScreenshotItem] {
        let descriptor = FetchDescriptor<ScreenshotItem>(
            predicate: #Predicate { $0.category == category && !$0.isDeleted }
        )
        return (try? modelContext.fetch(descriptor)) ?? []
    }
    
    func searchTemporary() -> [ScreenshotItem] {
        let descriptor = FetchDescriptor<ScreenshotItem>(
            predicate: #Predicate { $0.isTemporary && !$0.isDeleted }
        )
        return (try? modelContext.fetch(descriptor)) ?? []
    }
}
```

## 十、竞争优势总结

### 10.1 核心差异化

| 维度 | SnapSort AI | 竞品最佳 | 我们的优势 |
|------|------------|---------|-----------|
| 隐私 | 100%设备端 | Captr云端 | ✅ 零数据上传 |
| 价格 | $1.99/月 | $4.99/月 | ✅ 便宜60% |
| 免费额度 | 50张/月 | 10张/月 | ✅ 5倍免费 |
| 一次性买断 | $19.99 | $49.99+ | ✅ 便宜60% |
| App体积 | <30MB | 1.6GB(Screenshot PRO) | ✅ 不复制截图 |
| OCR精度 | Vision+规则引擎 | 单一OCR | ✅ 双重验证 |
| 临时清理 | 自动检测+提醒 | 无 | ✅ 独有功能 |
| 分类数量 | 13类 | 5-8类 | ✅ 更精细 |

### 10.2 护城河

1. **隐私品牌** — "100% On-Device"是2026年最强卖点，截图含密码/银行信息，用户极度重视
2. **价格壁垒** — $1.99/月是市场最低价，竞品难以跟进(已有高价用户基础)
3. **不复制截图** — 技术决策(仅存PHAsset引用)避免了Screenshot PRO的1.6GB膨胀问题
4. **临时截图自动清理** — 独有功能，竞品无此特性
5. **规则+ML双引擎** — OTP/QR/配送等高价值场景用规则精准识别，其他用ML模糊分类
